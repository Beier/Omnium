rule("No fly zone")
{
	event
	{
		Ongoing - Global;
	}
	
	conditions
	{
		Is true for any(All Players(Team(All)), Is in air(Current array element)) == true;
	}
	
	actions
	{
		Kill(Filtered array(All Players(Team(All)), Is in air(Current array element)), Null);
	}
}
